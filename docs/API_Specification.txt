RESTFUL API for the dashboard

1. Authentication API
URL: {BASE_URL:PORT}/auth
HTTP Methods: POST
INPUT: { 
			"auth_username":"USERNAME",
			"auth_password":"PASSWORD"
		}
OUTPUT: 
	1. if successful
		{
			"result":"success",
			"auth_username":"USERNAME"
			"auth_roles":["admin","user","developer"]
		}
	2. if unsuccessful
		{
			"result":"error",
			"error_message":"ERROR_MESSAGE"
		}

2. Data API
URL: {BASE_URL:PORT}/data

3. Comments API
URL: {BASE_URL:PORT}/comments
HTTP Headers:
	"auth":{
					"auth_username":"USERNAME",
					"auth_password":"PASSWORD"
			}
HTTP Methods: 
	1.GET
		URL: /comments/{id}
		OUTPUT: {
			"result":"success",
			"comment":"COMMENT"
			}
	2.POST
		URL: /comments/{id}
	3.PUT
	4.DELETE

			"id":"ID"	
		}
